public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        ListNode l = new ListNode();
        ListNode res = l;
        while(l1!=null||l2!=null){
            int val;
            int v1;
            int v2;
            if(l1==null){
                
                res.next = l2;
                return l.next;
            }
            if(l2==null){           
                res.next = l1;
                return l.next;
            }
            
            if(l1.val>l2.val){
                res.next = l2;
                l2 =l2.next;
                res = res.next;
            }else{
                res.next = l1;
                l1 =l1.next;
                res = res.next;
            }
        }

    }
