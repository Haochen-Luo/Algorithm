 public boolean hasCycle(ListNode head) {
         if (head == null) {
            return false;
        }
        ListNode slow = head;
        ListNode fast = head.next;
        while (slow != null && fast != null) {
            if (slow == fast) {
                return true;
            }
            if (slow.next == null) {
                return false;
            }
            slow = slow.next;
            for (int i = 0;i<2 ;i++ ) {
                if (fast.next ==null) {
                    return false;
                }
                fast = fast.next;
            }

        }
        return false;
    }
